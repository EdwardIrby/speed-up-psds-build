# Button

<Intro>
  Used to call attention, perform an action, or to navigate.
</Intro>

## Examples

```typescript
import Button from '@pluralsight/ps-design-system-button'
import React from 'react'

function Example() {
  return <div className="example-grid">
    <Button size={Button.sizes.large} appearance={Button.appearances.primary}>Large primary</Button>
    <Button size={Button.sizes.large} appearance={Button.appearances.secondary}>Large secondary</Button>
    <Button size={Button.sizes.large} appearance={Button.appearances.stroke}>Large stroke</Button>
    <Button size={Button.sizes.large} appearance={Button.appearances.flat}>Large flat</Button>

    <Button size={Button.sizes.medium} appearance={Button.appearances.primary}>Medium primary</Button>
    <Button size={Button.sizes.medium} appearance={Button.appearances.secondary}>Medium secondary</Button>
    <Button size={Button.sizes.medium} appearance={Button.appearances.stroke}>Medium stroke</Button>
    <Button size={Button.sizes.medium} appearance={Button.appearances.flat}>Medium flat</Button>

    <Button size={Button.sizes.small} appearance={Button.appearances.primary}>Medium primary</Button>
    <Button size={Button.sizes.small} appearance={Button.appearances.secondary}>Medium secondary</Button>
    <Button size={Button.sizes.small} appearance={Button.appearances.stroke}>Medium stroke</Button>
    <Button size={Button.sizes.small} appearance={Button.appearances.flat}>Medium flat</Button>

    <Button size={Button.sizes.xSmall} appearance={Button.appearances.primary}>xSmall primary</Button>
    <Button size={Button.sizes.xSmall} appearance={Button.appearances.secondary}>xSmall secondary</Button>
    <Button size={Button.sizes.xSmall} appearance={Button.appearances.stroke}>xSmall stroke</Button>
    <Button size={Button.sizes.xSmall} appearance={Button.appearances.flat}>xSmall flat</Button>
  </div>
}

export default Example
```
