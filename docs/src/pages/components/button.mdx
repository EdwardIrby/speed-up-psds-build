---
title: With react-router
description: Many users of this component are using it in conjunction with [`react-router`](https://github.com/ReactTraining/react-router/issues/1176) . If you'd like to do the same and use `Button` to trigger react-router links, you can follow this pattern.
---
import Button from '@pluralsight/ps-design-system-button'
import React from 'react'
import { BrowserRouter as Router, withRouter } from 'react-router-dom'

function Example() {
  // #1 Define your react-router-specific ButtonLink
  const ButtonLink = withRouter(props => (
    <Button
      {...props}
      onClick={evt => {
        evt.preventDefault();
        props.onClick && props.onClick(evt);
        props.history.push(props.to);
      }}
      href={props.to}
    />
  ))

  // #2 Invoke it like you'd usually use Link
  return <Router><ButtonLink to="/react-wonderland">React-router Link as DS button</ButtonLink></Router>
}

export default Example
