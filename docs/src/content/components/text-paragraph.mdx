### Colors

```typescript
import { P } from '@pluralsight/ps-design-system-text'
import React from 'react'

  return (
    <React.Fragment>
      <P color={P.colors.secondary}>
        Paragraph - Lorem ipsum dolor sit amet adipisicing elit, sed do eiusmod
```
