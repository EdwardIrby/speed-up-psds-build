# Label

<Intro>Labels are for all text except headings and paragraphs.</Intro>

<TableOfContents {...props} />

## Examples

### Sizes

```typescript
import { Label } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <React.Fragment>
      <div>
        <Label size={Label.sizes.large}>
          Large. The technology skills platform
        </Label>
      </div>
      <div>
        <Label size={Label.sizes.normal}>
          Normal. The technology skills platform
        </Label>
      </div>
      <div>
        <Label size={Label.sizes.small}>
          Small. The technology skills platform
        </Label>
      </div>
      <div>
        <Label size={Label.sizes.xsmall}>
          XSmall. The technology skills platform
        </Label>
      </div>
    </React.Fragment>
  )
}

export default Example
```

### Colors

Use the `color` prop to change the color.

```typescript
import { Label } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <>
    <div>
        <Label color={Caps.colors.primary}>
          Primary. The technology skills platform
        </Label>
      </div>
      <div>
        <Label color={Caps.colors.secondary}>
          Secondary. The technology skills platform
        </Label>
      </div>
    </>
  )
}

export default Example
```
