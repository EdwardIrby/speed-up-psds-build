# Heading

<Intro>
  Heading is used to create various levels of typographic hierarchies.
</Intro>

<TableOfContents {...props} />

## Examples

```typescript
import { Heading } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <React.Fragment>
      <Heading size={Heading.sizes.xLarge}>
        <h2>Heading (xLarge, primary)</h2>
      </Heading>

      <Heading size={Heading.sizes.xLarge} color={Heading.colors.secondary}>
        <h2>Heading (xLarge, secondary)</h2>
      </Heading>

      <Heading size={Heading.sizes.large}>
        <h2>Heading (large, primary)</h2>
      </Heading>

      <Heading size={Heading.sizes.large} color={Heading.colors.secondary}>
        <h2>Heading (large, secondary)</h2>
      </Heading>

      <Heading size={Heading.sizes.medium}>
        <h2>Heading (medium, primary)</h2>
      </Heading>

      <Heading size={Heading.sizes.medium} color={Heading.colors.secondary}>
        <h2>Heading (medium, secondary)</h2>
      </Heading>

      <Heading size={Heading.sizes.small}>
        <h2>Heading (small, primary)</h2>
      </Heading>

      <Heading size={Heading.sizes.small} color={Heading.colors.secondary}>
        <h2>Heading (small, secondary)</h2>
      </Heading>

      <Heading size={Heading.sizes.xSmall}>
        <h2>Heading (xSmall, primary)</h2>
      </Heading>

      <Heading size={Heading.sizes.xSmall} color={Heading.colors.secondary}>
        <h2>Heading (xSmall, secondary)</h2>
      </Heading>
    </React.Fragment>
  )
}

export default Example
```
