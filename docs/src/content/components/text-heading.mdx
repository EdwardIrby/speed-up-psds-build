# Heading

<Intro>
  Heading is used to create various levels of typographic hierarchies.
</Intro>

<TableOfContents {...props} />

## Examples

### Sizes

Use the `size` prop to set the size.

```typescript
import { Heading } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <React.Fragment>
      <Heading size={Heading.sizes.xlarge}>
        <h2>xlarge. The technology skills platform</h2>
      </Heading>

      <Heading size={Heading.sizes.large}>
        <h2>large. The technology skills platform</h2>
      </Heading>

      <Heading size={Heading.sizes.medium}>
        <h2>medium. The technology skills platform</h2>
      </Heading>
      <Heading size={Heading.sizes.small}>
        <h2>small. The technology skills platform</h2>
      </Heading>

      <Heading size={Heading.sizes.xsmall}>
        <h2>xsmall. The technology skills platform</h2>
      </Heading>
    </React.Fragment>
  )
}

export default Example
```

### Colors

Use the `color` prop to change the color.

```typescript
import { Heading } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <React.Fragment>
      <Heading color={Caps.colors.secondary}>
        <h2>The technology skills platform</h2>
      </Heading>
    </React.Fragment>
  )
}

export default Example
```
