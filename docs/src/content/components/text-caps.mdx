## Examples

### Sizes

Use the `size` prop to set the size.

```typescript
import { Caps } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <React.Fragment>
      <div>
        <Caps size={Caps.sizes.large}>
          Large. The technology skills platform
        </Caps>
      </div>
      <div>
        <Caps size={Caps.sizes.medium}>
          Medium. The technology skills platform
        </Caps>
      </div>
      <div>
        <Caps size={Label.sizes.small}>
          Small. The technology skills platform
        </Caps>
      </div>
    </React.Fragment>
  )
}

export default Example
```

### Colors

For reduced emphasis use `Cap.colors.secondary`.

```typescript
import { Caps } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <React.Fragment>
      <div>
        <Caps color={Caps.colors.primary}>Primary</Caps>
        <Caps color={Caps.colors.secondary}>Secondary</Caps>
      </div>
    </React.Fragment>
  )
}

export default Example
```
