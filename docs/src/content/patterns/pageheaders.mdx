## Basic header

A basic responsive page header with actions.

```typescript
import { breakpoints, layout } from '@pluralsight/ps-design-system-core'
import Button from '@pluralsight/ps-design-system-button'
import { PageWidthLayout } from '@pluralsight/ps-design-system-layout'
import SearchInput from '@pluralsight/ps-design-system-searchinput'
import { Heading } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <PageWidthLayout>
      <header className="header" id="basicheader">
        <Heading className="heading" size={Heading.sizes.xLarge}>
          <h1>Heading</h1>
        </Heading>

        <nav className="nav">
          <ul className="actions">
            <li>
              <SearchInput className="search" placeholder="Placeholder" />
            </li>
            <li>
              <Button appearance={Button.appearances.secondary}>Label</Button>
            </li>
            <li>
              <Button appearance={Button.appearances.secondary}>Label</Button>
            </li>
            <li>
              <Button appearance={Button.appearances.primary}>Label</Button>
            </li>
          </ul>
        </nav>
      </header>

      <style jsx>{`
        #basicheader {
          display: flex;
          flex-direction: column;
        }
        #basicheader .actions {
          display: flex;
          flex-direction: column;
          margin: 0;
          padding: 0;
          list-style: none;
        }
        #basicheader .heading {
          margin-bottom: ${layout.spacingLarge};
        }
        #basicheader .actions li {
          margin-bottom: ${layout.spacingXSmall};
          margin-right: 0;
        }
        #basicheader .search,
        #basicheader .actions button {
          width: 100%;
        }
        #basicheader .actions li:last-child {
          margin-bottom: 0;
        }

        @media screen and (min-width: ${breakpoints.small}) {
          #basicheader {
            flex-wrap: wrap;
          }
          #basicheader .heading {
            margin-bottom: ${layout.spacingLarge};
          }
          #basicheader .actions {
            flex-direction: row;
          }
          #basicheader .actions .search,
          #basicheader .actions button {
            width: initial;
          }
          #basicheader .actions li {
            margin-bottom: 0;
            margin-right: ${layout.spacingSmall};
          }
          #basicheader .actions li:last-child {
            margin-right: 0;
          }
        }

        @media screen and (min-width: ${breakpoints.large}) {
          #basicheader {
            flex-direction: row;
            align-items: flex-end;
          }
          #basicheader .heading {
            margin-bottom: 0;
            flex: 1;
          }
        }
      `}</style>
    </PageWidthLayout>
  )
}

export default Example
```

## Power header

A responsive page header with actions, tabs, and a breadcrumb link.

```typescript
import { breakpoints, layout } from '@pluralsight/ps-design-system-core'
import Badge from '@pluralsight/ps-design-system-badge'
import Breadcrumb from '@pluralsight/ps-design-system-breadcrumb'
import Button from '@pluralsight/ps-design-system-button'
import { PageWidthLayout } from '@pluralsight/ps-design-system-layout'
import Tab from '@pluralsight/ps-design-system-tab'
import { Heading } from '@pluralsight/ps-design-system-text'
import React from 'react'

const Example: React.FC = props => {
  return (
    <PageWidthLayout>
      <header className="header" id="powerheader">
        <div>
          <Breadcrumb>Label</Breadcrumb>
        </div>
        <div className="heading-actions">
          <Heading className="heading" size={Heading.sizes.large}>
            <h1>Heading</h1>
          </Heading>

          <ul className="actions">
            <li>
              <P>
                <strong className="label">Label: Status</strong>
                <br />
                <span className="subLabel">Sublabel: September 20, 2018</span>
              </P>
            </li>
            <li>
              <Button appearance={Button.appearances.primary}>Label</Button>
            </li>
          </ul>
        </div>

        <Tab.List>
          <Tab.ListItem id="menu1">Label</Tab.ListItem>
          <Tab.ListItem id="menu2">Label</Tab.ListItem>
          <Tab.ListItem id="menu3">Label</Tab.ListItem>
          <Tab.ListItem id="menu4">Label</Tab.ListItem>
          <Tab.ListItem id="menu5">Label</Tab.ListItem>
        </Tab.List>
      </header>

      <style jsx>{`
        #powerheader .heading-actions {
          display: flex;
          flex-direction: column;
          margin: 0;
          padding: 0;
          list-style: none;
        }
        #powerheader .heading {
          margin: 0 0 ${layout.spacingMedium} 0;
        }
        #powerheader .actions {
          display: flex;
          flex-direction: column;
          margin: 0;
          padding: 0;
          list-style: none;
        }

        #powerheader .actions p {
          line-height: var(--psTypeLineHeightTight);
        }
        #powerheader .actions .sublabel {
          color: ${colorsTextIcon.lowOnLight};
        }

        #powerheader .search,
        #powerheader .actions li {
          margin-bottom: ${layout.spacingXSmall};
        }
        #powerheader .search,
        #powerheader .actions button {
          width: 100%;
        }
        #powerheader .actions li:last-child {
          margin-bottom: 0;
        }

        @media screen and (min-width: ${breakpoints.medium}) {
          #powerheader .heading-actions {
            flex-direction: row;
            align-items: center;
          }
          #powerheader .actions {
            margin-bottom: 0;
            flex: 1
          }
          #powerheader .actions li {
            margin: 0;
            text-align: right;
          }
          #powerheader .actions button {
            width: initial;
          }
        }
      `}</style>
    </PageWidthLayout>
  )
}

export default Example
```
